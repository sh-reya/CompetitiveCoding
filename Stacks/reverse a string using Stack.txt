
class Solution {
    
    public void reverse(String str)
    {
        Stack<Character> s=new Stack<Character>();
        for(int i=0; i<str.length(); i++)
        {
            char ch=str.charAt(i);
            s.push(ch);
        }
        for(int i=0; i<str.length(); i++)
        {
           // s.pop();
            System.out.print(s.pop());
        }
        System.out.println();
       //your code here
    }

}
