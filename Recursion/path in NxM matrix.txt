import java.util.*;
public class Main
{
    static int path(int a, int b)
    {
        if(a==1 || b==1)
        {
            return 1;
        }
        
        return (path(a,b-1)+path(a-1,b));
        
        
    }
    
    
	public static void main(String[] args) 
	{
		Scanner s=new Scanner(System.in);
		int a=s.nextInt();
		int b=s.nextInt();
		System.out.println(path(a,b));
	}
}
