import java.util.*;
public class Main
{
    static int fastpow(int a, int b)
    {
        if(b==0)
        {
            return 1;
        }
        if(b%2==0)
        {
            return fastpow(a*a,b/2);
        }
        
        
        return a*fastpow(a,b-1);
        
        
        
        
        
        
    }
	public static void main(String[] args) 
	{
		Scanner s=new Scanner(System.in);
		int a=s.nextInt();
		int b=s.nextInt();
		System.out.println(fastpow(a,b));
	}
}
