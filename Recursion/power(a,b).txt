import java.util.*;
public class Main
{
    static int pow(int a , int b)
    {
        if(b==0)
        {
            return 1;
        }
        
        
        return (a*pow(a , b-1));
        
        
    }
	public static void main(String[] args) 
	{
		Scanner s=new Scanner(System.in);
		int a=s.nextInt();
		int b=s.nextInt();
		System.out.println(pow(a,b));
	}
}
