class ReverseLL
{
    // This function should reverse linked list and return
    // head of the modified linked list.
    Node reverseList(Node head)
    {
        Node t1=head;
        Node t2;
        Node prev=null;
        
        while(t1!=null)
        {
            t2=t1.next;
            t1.next=prev;
            prev=t1;
            t1=t2;
        }
        
        return prev;
    }
}
