
class Palindrome
{
    // Function to check if linked list is palindrome
    boolean isPalindrome(Node head) 
    {
        Node t1=head;
        Node t2=head;
        boolean p=true;
        Stack<Integer>s=new Stack<Integer>();
        
        while(t1!=null)
        {
            s.push(t1.data);
            t1=t1.next;
        }
        
        while(t2!=null)
        {
            int i=s.pop();
            if(t2.data==i)
            {
                p=true;
            }
            
            else
            {
                p=false;
                break;
            }
            t2=t2.next;
        }
        
        return p;

    }    
}
