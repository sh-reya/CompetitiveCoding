class GfG
{
    public static Node reverse(Node node, int k)
    {
       
        Node current=node;
        Node next = null; 
        Node prev = null; 
        
        int count=0;
        
        
        while (count < k && current != null)  
       { 
           next = current.next; 
           current.next = prev; 
           prev = current; 
           current = next; 
           count++; 
       } 
       
        if (next != null)  
          node.next = reverse(next, k); 
          
         return prev;
  
  
        //Your code here
    }
}
